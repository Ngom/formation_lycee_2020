\section{Un algorithme}
\begin{algorithm}
$n \leftarrow 0 $ \\
$u \leftarrow 2 $ \\
$k \leftarrow 45 $ \\
\Tq{$u<k$}{
$u \leftarrow 1,04 \times u $ \\
$n \leftarrow n+1 $ \\
}
\end{algorithm}
